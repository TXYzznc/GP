# 棋子系统类图

```mermaid
---
config:
    classDiagram:
        nodeSpacing: 60
        rankSpacing: 80
        curve: linear
---
classDiagram
    class ChessEntity {
        +ChessAI m_AI
        +CombatController m_Combat
        +IChessSkill m_NormalAttack
        +IChessSkill[] m_Skills
        +Transform m_Transform
        +Animator m_Animator
        +Initialize()
        +Tick()
        +TakeDamage(damage, isCritical)
        +Die()
    }
    
    class ChessAI {
        +ChessEntity m_Entity
        +bool m_IsInCombat
        +FindTarget() ChessEntity
        +Attack(target)
        +Tick()
        +UpdateAggression()
    }
    
    class CombatController {
        +ChessEntity m_Entity
        +IChessSkill m_CurrentSkill
        +OnAttackExecute(skillId)
        +OnSkillExecute(skillId)
        +PlayAttackAnimation()
    }
    
    class IChessSkill {
        <<interface>>
        +ExecuteAttack(caster, target)*
        +CalculateDamage(caster)*
        +GetSkillConfig()*
    }
    
    class ChessNormalAttackBase {
        <<abstract>>
        +SkillConfig m_Config
        +ExecuteAttack(caster, target)*
        +CalculateDamage(caster, isCritical)*
    }
    
    class HitDetectorFactory {
        <<static>>
        +GetDetector(type)$ IHitDetector
    }
    
    class IHitDetector {
        <<interface>>
        +Execute(context)*
    }
    
    class HitDetectorBase {
        <<abstract>>
        +ApplyDamage(context)
        +EffectExecutor.ApplyBuffsOnHit()
    }
    
    class InstantDetector {
        +Execute(context)
    }
    
    class MeleeDetector {
        +Execute(context)
    }
    
    class ProjectileDetector {
        +Execute(context)
    }
    
    class HitContext {
        +ChessEntity Attacker
        +ChessEntity LockedTarget
        +double BaseDamage
        +bool IsCritical
        +SkillConfig SkillConfig
        +Action OnHitCallback
    }
    
    ChessEntity --> ChessAI: owns
    ChessEntity --> CombatController: owns
    ChessEntity --> IChessSkill: has
    CombatController --> IChessSkill: executes
    ChessAI --> ChessEntity: targets
    
    IChessSkill <|.. ChessNormalAttackBase
    
    ChessNormalAttackBase --> HitDetectorFactory: uses
    HitDetectorFactory --> IHitDetector: creates
    
    IHitDetector <|.. HitDetectorBase
    HitDetectorBase <|.. InstantDetector
    HitDetectorBase <|.. MeleeDetector
    HitDetectorBase <|.. ProjectileDetector
    
    HitDetectorBase --> HitContext: uses
```

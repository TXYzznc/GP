## 任务概要

优化战斗相机系统，实现以下功能：
1. 相机平滑移动到战斗位置，与场景加载并行
2. 添加战斗视角模式（只支持缩放）
3. 进入战斗时锁定视角，离开战斗时恢复

## 功能需求

### 1. 相机平滑移动
- 进入战斗准备阶段时，相机平滑移动到 CameraAnchor 位置
- 移动时间约 0.8 秒
- 相机移动与战斗场景加载并行执行（先开始移动，再加载场景）

### 2. 战斗视角模式
- 在 `CameraViewMode` 枚举中添加 `Combat` 模式
- 战斗视角特性：
  - 相机位置固定，不跟随玩家
  - 不支持鼠标旋转
  - 只支持滚轮缩放
  - 缩放时相机沿视线方向前后移动

### 3. 视角锁定与恢复
- 进入战斗准备阶段：
  - 缓存当前视角模式
  - 切换到战斗视角
  - 锁定视角切换（玩家无法通过按键切换视角）
- 离开战斗阶段：
  - 解锁视角切换
  - 恢复到战斗前的视角模式

private void HandleCombatMode()
{

/// <summary>
/// 更新战斗模式下的相机缩放
/// 沿相机视线方向前后移动
/// </summary>
private void UpdateCombatCameraZoom()
{
    // 平滑过渡到目标距离
    m_CurrentDistance = Mathf.SmoothDamp(
        m_CurrentDistance,
        m_TargetDistance,
        ref m_DistanceVelocity,
        zoomSmoothTime
    );

    // 沿相机当前朝向的反方向移动（后退）
    Vector3 direction = -transform.forward;
    Vector3 targetPos = transform.position + direction * (m_CurrentDistance - m_OccludedDistance);

    // 更新相机位置
    if (m_Camera != null)
    {
        m_Camera.transform.position = targetPos;
    }
}

#### 修改 4：添加平滑移动方法（约第 550 行之后）

```csharp
/// <summary>
/// 平滑移动到目标位置和旋转
/// </summary>
/// <param name="targetPosition">目标位置</param>
/// <param name="targetRotation">目标旋转</param>
/// <param name="duration">移动时间（秒）</param>
public void SmoothMoveTo(Vector3 targetPosition, Quaternion targetRotation, float duration)
{
    DebugEx.LogModule("ThirdPersonCamera", 
        $"开始平滑移动: 目标位置={targetPosition}, 目标旋转={targetRotation.eulerAngles}, 时长={duration}秒");

    m_IsSmoothMoving = true;
    m_TargetPosition = targetPosition;
    m_TargetRotation = targetRotation;
    m_SmoothMoveTime = duration;
    m_SmoothMoveElapsed = 0f;
}

/// <summary>
/// 更新平滑移动
/// </summary>
private void UpdateSmoothMove()
{
    m_SmoothMoveElapsed += Time.deltaTime;
    float t = Mathf.Clamp01(m_SmoothMoveElapsed / m_SmoothMoveTime);

    // 使用 SmoothStep 曲线使移动更自然
    t = Mathf.SmoothStep(0f, 1f, t);

    // 插值位置和旋转
    transform.position = Vector3.Lerp(transform.position, m_TargetPosition, t);
    transform.rotation = Quaternion.Slerp(transform.rotation, m_TargetRotation, t);

    // 移动完成
    if (m_SmoothMoveElapsed >= m_SmoothMoveTime)
    {
        transform.position = m_TargetPosition;
        transform.rotation = m_TargetRotation;
        m_IsSmoothMoving = false;

        DebugEx.LogModule("ThirdPersonCamera", "平滑移动完成");
    }
}
```

/// <summary>
/// 设置视角模式
/// </summary>
public void SetViewMode(CameraViewMode mode)
{
    // ⭐ 检查视角是否被锁定
    if (m_IsViewModeLocked && mode != viewMode)
    {
        DebugEx.WarningModule("ThirdPersonCamera", $"视角模式已锁定，无法切换到 {mode}");
        return;
    }

    if (viewMode != mode)
    {
        DebugEx.LogModule("ThirdPersonCamera", $"切换视角模式: {viewMode} -> {mode}");
        viewMode = mode;
    }
}
```

private void SetupCombatCamera()

    ThirdPersonCamera cameraController = CameraRegistry.ThirdPersonCamera;
    if (cameraController == null)
    {
        DebugEx.ErrorModule("CombatPreparationState", "未找到第三人称相机控制器");
        return;

    // 缓存当前视角模式
    cameraController.CacheCurrentViewMode();
    DebugEx.LogModule("CombatPreparationState", "已缓存战斗前的视角模式");

    // 切换到战斗视角
    cameraController.SetViewMode(CameraViewMode.Combat);
    DebugEx.LogModule("CombatPreparationState", "已切换到战斗视角");

    // 锁定视角切换
    cameraController.SetViewModeLocked(true);
    DebugEx.LogModule("CombatPreparationState", "已锁定视角切换");
}

/// <summary>
/// 生成战斗场地（带溶解过渡）

private async void SpawnBattleArena()
{
    if (BattleArenaManager.Instance != null)
    {
        // ⭐ 先开始移动相机（异步，不等待）
        StartCameraSmoothMove();

        // 在玩家当前位置附近生成战斗场地
        Vector3 spawnPos = Vector3.zero;
        if (PlayerCharacterManager.Instance != null &&
            PlayerCharacterManager.Instance.CurrentPlayerCharacter != null)
        {
            await BattleArenaManager.Instance.SpawnArenaAsync(PlayerCharacterManager.Instance.CurrentPlayerCharacter.transform);
        }

        // 播放溶解过渡效果（从探索场景显示战斗场地）
        var arena = BattleArenaManager.Instance.CurrentArena;
        if (arena != null)
        {
            await DissolveTransitionManager.Instance.TransitionToBattle(arena);
        }

        // 加载敌人波次配置（使用保存的战斗数据）
        LoadEnemyWaveConfig();
    }
    else
    {
        DebugEx.WarningModule("CombatPreparationState", "BattleArenaManager 未初始化");
    }
}

/// <summary>
/// 开始相机平滑移动到 CameraAnchor
/// </summary>
private void StartCameraSmoothMove()
{
    DebugEx.LogModule("CombatPreparationState", "开始相机平滑移动...");

    // 获取战斗场地
    var arena = BattleArenaManager.Instance?.CurrentArena;
    if (arena == null)
    {
        DebugEx.WarningModule("CombatPreparationState", "战斗场地不存在，跳过相机移动");
        return;
    }

    // 查找 CameraAnchor
    Transform cameraAnchor = arena.transform.Find("CameraAnchor");
    if (cameraAnchor == null)
    {
        DebugEx.WarningModule("CombatPreparationState", "未找到 CameraAnchor，跳过相机移动");
        return;
    }

    // 获取相机控制器
    ThirdPersonCamera cameraController = CameraRegistry.ThirdPersonCamera;
    if (cameraController == null)
    {
        DebugEx.ErrorModule("CombatPreparationState", "未找到第三人称相机控制器");
        return;
    }

    // 开始平滑移动（0.8秒）
    cameraController.SmoothMoveTo(cameraAnchor.position, cameraAnchor.rotation, 0.8f);

    DebugEx.LogModule("CombatPreparationState",
        $"相机开始移动到 CameraAnchor: Pos={cameraAnchor.position}, Rot={cameraAnchor.rotation.eulerAngles}");
}
```

## 完整实现

```csharp
using UnityEngine;
using System;

namespace AAAGame.Combat
{
    /// <summary>
    /// 伤害计算器：统一处理所有伤害计算逻辑
    /// 包含：物理伤害、魔法伤害、真实伤害、护甲减伤、破甲增伤、暴击
    /// </summary>
    public static class DamageCalculator
    {
        // ===== 护甲参数 =====
        private const float ARMOR_LOW_THRESHOLD = 100f;
        private const float ARMOR_LOW_RATE = 0.03f;        // 3%
        private const float ARMOR_HIGH_BASE = 0.03f;       // 3%
        private const float ARMOR_HIGH_RATE = 0.01f;       // 1%
        private const float ARMOR_MAX_REDUCTION = 0.85f;   // 最多85%减伤
        
        // ===== 法抗参数 =====
        private const float MAGICRESIST_LOW_THRESHOLD = 80f;
        private const float MAGICRESIST_LOW_RATE = 0.035f; // 3.5%
        private const float MAGICRESIST_HIGH_BASE = 0.025f; // 2.5%
        private const float MAGICRESIST_HIGH_RATE = 0.008f; // 0.8%
        private const float MAGICRESIST_MAX_REDUCTION = 0.80f; // 最多80%减伤
        
        // ===== 破甲参数 =====
        private const float NEGATIVE_ARMOR_RATE = 0.015f;     // 破甲系数 = 护甲系数×50%
        private const float NEGATIVE_MAGICRESIST_RATE = 0.018f; // 破法系数更强
        private const float ARMOR_MIN_DAMAGE_PASSTHROUGH = 0.15f; // 最小伤害穿透15%
        private const float MAX_POSITIVE_ARMOR_INCREASE = 0.30f; // 最多增伤30%
        private const float MIN_ARMOR_LIMIT = -200f;           // 护甲下限
        
        // ===== 暴击参数 =====
        private const float CRIT_RATE_CAP = 0.75f;         // 暴击率上限75%
        private const float CRIT_BASE_MULTIPLIER = 1.5f;   // 基础倍数1.5倍
        
        // ===== 伤害类型 =====
        public enum DamageType
        {
            Physical = 1,  // 物理伤害
            Magic = 2,     // 魔法伤害
            True = 3       // 真实伤害
        }

        #region 公式1-3：基础伤害计算

        /// <summary>
        /// 公式1：物理伤害 = 攻击力 × 技能系数 × (1 + 增伤%) × 暴击倍数
        /// </summary>
        public static double CalculatePhysicalDamage(double attackPower, double skillCoefficient,
            double damageIncreaseRatio = 0, double critRate = 0, double critPower = 0)
        {
            // 1. 基础伤害
            double baseDamage = attackPower * skillCoefficient;

            // 2. 应用增伤Buff
            double damageWithBuff = baseDamage * (1 + Mathf.Clamp((float)damageIncreaseRatio, 0, 2f));

            // 3. 应用暴击
            double damageAfterCrit = ApplyCritical(damageWithBuff, critRate, critPower);

            return damageAfterCrit;
        }

        /// <summary>
        /// 公式2：魔法伤害
        /// </summary>
        public static double CalculateMagicDamage(double magicPower, double skillCoefficient,
            double damageIncreaseRatio = 0, double critRate = 0, double critPower = 0)
        {
            // 逻辑与物理伤害相同
            double baseDamage = magicPower * skillCoefficient;
            double damageWithBuff = baseDamage * (1 + Mathf.Clamp((float)damageIncreaseRatio, 0, 2f));
            double damageAfterCrit = ApplyCritical(damageWithBuff, critRate, critPower);

            return damageAfterCrit;
        }

        /// <summary>
        /// 公式3：真实伤害 = 真伤力 × 技能系数 × (1 + 增伤%) × 暴击倍数
        /// 真实伤害不受护甲/法抗影响
        /// </summary>
        public static double CalculateTrueDamage(double truePower, double skillCoefficient,
            double damageIncreaseRatio = 0, double critRate = 0, double critPower = 0)
        {
            double baseDamage = truePower * skillCoefficient;
            double damageWithBuff = baseDamage * (1 + Mathf.Clamp((float)damageIncreaseRatio, 0, 2f));
            double damageAfterCrit = ApplyCritical(damageWithBuff, critRate, critPower);

            return damageAfterCrit;
        }

        #endregion

        #region 公式4-6：防御减伤和破甲

        /// <summary>
        /// 公式4：护甲减伤系数（支持负数护甲 = 破甲增伤）
        /// 
        /// 正数护甲：减伤（两段式递减）
        /// - 低阈值段：线性递减
        /// - 高阈值段：递减
        /// 
        /// 负数护甲：增伤（破甲）
        /// - 增伤系数 = 护甲系数 × 50%（平缓）
        /// - 最多增伤 30%
        /// </summary>
        public static float CalculateArmorReduction(float armor)
        {
            // 硬下限：护甲不能低于-200
            armor = Mathf.Max(armor, MIN_ARMOR_LIMIT);

            // ===== 正数护甲：减伤 =====
            if (armor > 0)
            {
                float reduction;

                if (armor <= ARMOR_LOW_THRESHOLD)
                {
                    // 低护甲段：线性递减
                    reduction = armor * ARMOR_LOW_RATE;
                }
                else
                {
                    // 高护甲段：递减效应
                    float excessArmor = armor - ARMOR_LOW_THRESHOLD;
                    reduction = ARMOR_HIGH_BASE + (excessArmor * ARMOR_HIGH_RATE);
                }

                // 限制最大减伤
                reduction = Mathf.Min(reduction, ARMOR_MAX_REDUCTION);
                return 1f - reduction;
            }

            // ===== 负数护甲：增伤（破甲）=====
            else if (armor < 0)
            {
                // 破甲增伤 = -护甲 × ARMOR_LOW_RATE × NEGATIVE_ARMOR_RATE
                float damageIncrease = -armor * ARMOR_LOW_RATE * NEGATIVE_ARMOR_RATE;

                // 限制最大增伤（防止伤害爆炸）
                damageIncrease = Mathf.Min(damageIncrease, MAX_POSITIVE_ARMOR_INCREASE);

                return 1f + damageIncrease;
            }

            // armor == 0
            return 1f;
        }

        /// <summary>
        /// 公式5：法抗减伤系数（支持负数法抗 = 破法增伤）
        /// 参数比护甲更弱，鼓励魔法输出
        /// </summary>
        public static float CalculateMagicResistReduction(float magicResist)
        {
            // 硬下限
            magicResist = Mathf.Max(magicResist, MIN_ARMOR_LIMIT);

            // ===== 正数法抗：减伤 =====
            if (magicResist > 0)
            {
                float reduction;

                if (magicResist <= MAGICRESIST_LOW_THRESHOLD)
                {
                    reduction = magicResist * MAGICRESIST_LOW_RATE;
                }
                else
                {
                    float excessMR = magicResist - MAGICRESIST_LOW_THRESHOLD;
                    reduction = MAGICRESIST_HIGH_BASE + (excessMR * MAGICRESIST_HIGH_RATE);
                }

                reduction = Mathf.Min(reduction, MAGICRESIST_MAX_REDUCTION);
                return 1f - reduction;
            }

            // ===== 负数法抗：增伤（破法）=====
            else if (magicResist < 0)
            {
                // 破法增伤 = -法抗 × MAGICRESIST_LOW_RATE × NEGATIVE_MAGICRESIST_RATE
                float damageIncrease = -magicResist * MAGICRESIST_LOW_RATE * NEGATIVE_MAGICRESIST_RATE;

                // 法抗的破法上限稍高（因为法抗本身就弱）
                damageIncrease = Mathf.Min(damageIncrease, 0.35f);

                return 1f + damageIncrease;
            }

            return 1f;
        }

        #endregion

        #region 公式6：暴击伤害

        /// <summary>
        /// 公式6：暴击伤害计算
        /// 暴击倍数 = 1.5 + 爆伤值（至少1.5倍）
        /// 
        /// 示例：
        /// - 20%暴击率，0.5爆伤 → 平均伤害 = 100 × (0.8×1.0 + 0.2×2.0) = 120
        /// - 50%暴击率，1.0爆伤 → 平均伤害 = 100 × (0.5×1.0 + 0.5×2.5) = 175
        /// </summary>
        private static double ApplyCritical(double baseDamage, double critRate, double critPower)
        {
            // 暴击率上限
            float clampedCritRate = Mathf.Clamp((float)critRate, 0, CRIT_RATE_CAP);

            // 判定暴击
            bool isCrit = UnityEngine.Random.Range(0f, 1f) < clampedCritRate;

            if (isCrit)
            {
                // 爆伤倍数 = 基础倍数 + 额外倍数（至少1.5倍）
                double critMultiplier = CRIT_BASE_MULTIPLIER + Mathf.Clamp((float)critPower, 0, 1.5f);
                return baseDamage * critMultiplier;
            }

            return baseDamage;
        }

        #endregion

        #region 公式7-11：完整伤害计算流程

        /// <summary>
        /// 公式11：完整物理伤害计算（从攻击到最终伤害）
        /// 流程：基础伤害 → 增伤Buff → 暴击 → 护甲减伤 → 护盾抵消 → 最小伤害穿透
        /// </summary>
        public static double CalculateFinalPhysicalDamage(
            double attackPower, double skillCoefficient, double targetArmor,
            double maxHealth, double damageIncreaseRatio = 0,
            double critRate = 0, double critPower = 0, double shieldValue = 0)
        {
            // Step 1: 基础伤害
            double baseDamage = attackPower * skillCoefficient;

            // Step 2: 应用增伤Buff
            double damageWithBuff = baseDamage * (1 + Mathf.Clamp((float)damageIncreaseRatio, 0, 2f));

            // Step 3: 应用暴击
            double damageAfterCrit = ApplyCritical(damageWithBuff, critRate, critPower);

            // Step 4: 护甲减伤
            float armorReduction = CalculateArmorReduction((float)targetArmor);
            double damageAfterArmor = damageAfterCrit * armorReduction;

            // Step 5: 护盾抵消
            double damageAfterShield = ApplyShield(damageAfterArmor, shieldValue);

            // Step 6: 最小伤害穿透（至少15%最大血量）
            double minDamage = maxHealth * ARMOR_MIN_DAMAGE_PASSTHROUGH;
            double finalDamage = Mathf.Max((float)damageAfterShield, (float)minDamage);

            return finalDamage;
        }

        /// <summary>
        /// 完整魔法伤害计算
        /// </summary>
        public static double CalculateFinalMagicDamage(
            double magicPower, double skillCoefficient, double targetMagicResist,
            double maxHealth, double damageIncreaseRatio = 0,
            double critRate = 0, double critPower = 0, double shieldValue = 0)
        {
            double baseDamage = magicPower * skillCoefficient;
            double damageWithBuff = baseDamage * (1 + Mathf.Clamp((float)damageIncreaseRatio, 0, 2f));
            double damageAfterCrit = ApplyCritical(damageWithBuff, critRate, critPower);

            float mrReduction = CalculateMagicResistReduction((float)targetMagicResist);
            double damageAfterMR = damageAfterCrit * mrReduction;

            double damageAfterShield = ApplyShield(damageAfterMR, shieldValue);

            double minDamage = maxHealth * ARMOR_MIN_DAMAGE_PASSTHROUGH;
            double finalDamage = Mathf.Max((float)damageAfterShield, (float)minDamage);

            return finalDamage;
        }

        /// <summary>
        /// 完整真实伤害计算（不受防御影响）
        /// </summary>
        public static double CalculateFinalTrueDamage(
            double truePower, double skillCoefficient,
            double maxHealth, double damageIncreaseRatio = 0,
            double critRate = 0, double critPower = 0, double shieldValue = 0)
        {
            double baseDamage = truePower * skillCoefficient;
            double damageWithBuff = baseDamage * (1 + Mathf.Clamp((float)damageIncreaseRatio, 0, 2f));
            double damageAfterCrit = ApplyCritical(damageWithBuff, critRate, critPower);

            // 真实伤害不受防御影响，直接护盾抵消
            double damageAfterShield = ApplyShield(damageAfterCrit, shieldValue);

            // 真实伤害也有最小穿透（可选，一般不需要）
            return damageAfterShield;
        }

        #endregion

        #region 公式7：护盾抵消

        /// <summary>
        /// 公式7：护盾吸收伤害
        /// 护盾优先级最高，完全抵消伤害
        /// </summary>
        private static double ApplyShield(double damage, double shieldValue)
        {
            if (shieldValue > 0)
            {
                if (damage <= shieldValue)
                {
                    // 伤害完全抵消
                    return 0;
                }
                else
                {
                    // 护盾不足，扣血
                    return damage - shieldValue;
                }
            }

            return damage;
        }

        #endregion

        #region 调试和日志

        /// <summary>
        /// 详细的伤害计算日志（调试用）
        /// </summary>
        public static void LogDamageCalculation(
            string attackerName, string targetName,
            double baseDamage, double finalDamage,
            double targetDefense, float defenseReduction, DamageType damageType)
        {
            string defenseType = damageType == DamageType.Physical ? "护甲" : "法抗";
            string log = $"[伤害计算] {attackerName} → {targetName}\n" +
                        $"  伤害类型: {damageType}\n" +
                        $"  基础伤害: {baseDamage:F1}\n" +
                        $"  目标{defenseType}: {targetDefense:F0}\n" +
                        $"  {defenseType}减伤: {(1 - defenseReduction) * 100:F1}%\n" +
                        $"  最终伤害: {finalDamage:F1}";

            DebugEx.LogModule("DamageCalculator", log);
        }

        #endregion

        #region 单元测试用例

        /// <summary>
        /// 单元测试：验证公式正确性
        /// </summary>
        public static void RunTests()
        {
            // 测试1：基础伤害
            double test1 = CalculatePhysicalDamage(100, 1.0);
            Debug.Assert(Mathf.Abs((float)test1 - 100f) < 0.01f, $"测试1失败: {test1}");

            // 测试2：+100护甲减伤应该是3%
            float test2 = CalculateArmorReduction(100);
            Debug.Assert(Mathf.Abs(test2 - 0.97f) < 0.01f, $"测试2失败: {test2}");

            // 测试3：-100破甲增伤应该是1.5%
            float test3 = CalculateArmorReduction(-100);
            Debug.Assert(Mathf.Abs(test3 - 1.015f) < 0.005f, $"测试3失败: {test3}");

            // 测试4：-200护甲达到下限，增伤3%
            float test4 = CalculateArmorReduction(-200);
            Debug.Assert(test4 <= 1.031f, $"测试4失败: {test4}");

            DebugEx.LogModule("DamageCalculator", "✓ 所有单元测试通过");
        }

        #endregion
    }
}
```

## 使用示例

### 示例1：计算物理攻击伤害

```csharp
public class PhysicalAttack
{
    public void ExecuteAttack(ChessEntity attacker, ChessEntity target)
    {
        double damage = DamageCalculator.CalculateFinalPhysicalDamage(
            attackPower: attacker.Attribute.Attack,
            skillCoefficient: 1.0,
            targetArmor: target.Attribute.Armor,
            maxHealth: target.Attribute.MaxHealth,
            damageIncreaseRatio: 0.25,  // 有25%增伤Buff
            critRate: 0.2,               // 20%暴击率
            critPower: 0.5,              // 50%爆伤
            shieldValue: target.GetShieldValue()
        );

        target.TakeDamage((int)damage);
        
        DamageCalculator.LogDamageCalculation(
            attacker.Name, target.Name,
            attacker.Attribute.Attack * 1.0,
            damage,
            target.Attribute.Armor,
            DamageCalculator.CalculateArmorReduction(target.Attribute.Armor),
            DamageCalculator.DamageType.Physical
        );
    }
}
```

### 示例2：破甲技能

```csharp
public class BreakArmorSkill : ChessSkillBase
{
    public override void ExecuteSkill(ChessEntity caster, ChessEntity target)
    {
        // 1. 造成伤害
        double damage = DamageCalculator.CalculateFinalPhysicalDamage(
            caster.Attribute.Attack, 1.2, target.Attribute.Armor,
            target.Attribute.MaxHealth);
        
        target.TakeDamage((int)damage);
        
        // 2. 应用破甲
        target.Attribute.ApplyArmorModifier(-50);  // -50护甲
        
        DebugEx.LogModule("BreakArmorSkill",
            $"对{target.Name}施加破甲，当前护甲: {target.Attribute.GetFinalArmor()}");
    }
}
```

### 示例3：多段伤害（如连击技能）

```csharp
public class MultiStrikeAttack : ChessSkillBase
{
    private const int STRIKE_COUNT = 5;
    
    public override void ExecuteSkill(ChessEntity caster, ChessEntity target)
    {
        double totalDamage = 0;

        for (int i = 0; i < STRIKE_COUNT; i++)
        {
            double singleDamage = DamageCalculator.CalculateFinalPhysicalDamage(
                caster.Attribute.Attack * 0.4,  // 每下40%攻击力
                1.0,
                target.Attribute.Armor,
                target.Attribute.MaxHealth
            );

            totalDamage += singleDamage;
            target.TakeDamage((int)singleDamage);
        }

        DebugEx.LogModule("MultiStrikeAttack",
            $"共造成{totalDamage:F0}伤害");
    }
}
```

## 与ChessAttribute的集成

```csharp
public class ChessAttribute : MonoBehaviour
{
    private float m_BaseArmor;
    private float m_ArmorModifier;  // Buff和破甲导致的修改

    /// <summary>
    /// 获取最终护甲值（包括所有修改）
    /// </summary>
    public float GetFinalArmor()
    {
        float finalArmor = m_BaseArmor + m_ArmorModifier;
        return Mathf.Max(finalArmor, -200f);  // 下限-200
    }

    /// <summary>
    /// 应用护甲修改（用于破甲Buff）
    /// </summary>
    public void ApplyArmorModifier(float modifier)
    {
        m_ArmorModifier += modifier;
        DebugEx.LogModule("ChessAttribute",
            $"护甲修改: {modifier:+0;-0}, 最终护甲: {GetFinalArmor():F0}");
    }
}
```
